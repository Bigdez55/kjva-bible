# Global State Management Agent

## Overview
Replace top-level mutable globals with a thread-safe singleton repository. Centralizes all note/trash mutations into a single authoritative source with proper access control and notification.

## Tier & Priority
- **Tier:** 2 (Feature Integrity & Architecture)
- **Priority:** P1 (Correctness)
- **Estimated Duration:** 6–8 hours
- **Blocker:** YES (for ADDNOTE-REFACTOR)

## Scope

### Issue: Global Mutable Arrays with No Encapsulation
**Files:** 
- `Apollo16/VIEW CONTROLLERS/AllNotes.swift` (top-level scope, lines 1-30)
- `Apollo16/VIEW CONTROLLERS/TrashedNotes.swift` (references)
- `Apollo16/VIEW CONTROLLERS/AddNote.swift` (references)

**Current Pattern:**
```swift

var array_notes        = [Note]()
var array_trashedNotes = [Note]()
```

**Problems:**
1. No access control — any file can mutate directly
2. No synchronization — race conditions if accessed from multiple threads
3. No notification — callers don't know when to reload UI
4. 12+ mutation sites scattered across 4 files
5. No single source of truth — state is fragmented
6. Duplicated save logic (resign path vs back-press path)

### Solution: NoteRepository Singleton

**New File:** `Apollo16/GLOBALS/MANAGERS/NoteRepository.swift`

```swift
import Foundation

/// Thread-safe, authoritative repository for notes and trash.
/// All mutations post NotificationCenter notifications.
/// Access is read-through property or mutation methods.
class NoteRepository: NSObject {
    
    static let shared = NoteRepository()
    
    // MARK: - Notifications
    static let notesDidChangeNotification = NSNotification.Name("NoteRepository.notesDidChange")
    static let trashedNotesDidChangeNotification = NSNotification.Name("NoteRepository.trashedNotesDidChange")
    
    // MARK: - Private Storage
    private let queue = DispatchQueue(label: "com.apollo16.note-repository", attributes: .concurrent)
    private var _notes: [Note] = []
    private var _trashedNotes: [Note] = []
    
    // MARK: - Public Read-Only Access
    var notes: [Note] {
        queue.sync { _notes }
    }
    
    var trashedNotes: [Note] {
        queue.sync { _trashedNotes }
    }
    
    // MARK: - Initialization
    private override init() {
        super.init()
        loadNotesFromDisk()
    }
    
    // MARK: - Loading
    private func loadNotesFromDisk() {
        queue.async(flags: .barrier) { [weak self] in
            let loaded = DefaultsUtility.shared.getNotes()
            let trashed = DefaultsUtility.shared.getTrashedNotes()
            
            self?._notes = loaded
            self?._trashedNotes = trashed
            
            // Enforce 15-day auto-delete
            self?.enforceTrashedNoteExpiry()
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    // MARK: - Note Mutations
    
    /// Add a new note or update an existing one.
    func saveNote(_ note: Note) {
        queue.async(flags: .barrier) { [weak self] in
            if let index = self?._notes.firstIndex(where: { $0.uid == note.uid }) {
                self?._notes[index] = note
            } else {
                self?._notes.append(note)
            }
            
            // Persist to disk
            DefaultsUtility.shared.saveNote(note)
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
            }
        }
    }
    
    /// Delete a note (move to trash).
    func trashNote(uid: String) {
        queue.async(flags: .barrier) { [weak self] in
            guard let index = self?._notes.firstIndex(where: { $0.uid == uid }) else { return }
            
            var note = self!._notes.remove(at: index)
            note.isDeleted = true
            note.deletedDate = Date()
            
            self?._trashedNotes.append(note)
            
            // Persist
            DefaultsUtility.shared.saveNote(note)
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    /// Delete multiple notes (move to trash).
    func trashNotes(uids: [String]) {
        queue.async(flags: .barrier) { [weak self] in
            for uid in uids {
                guard let index = self?._notes.firstIndex(where: { $0.uid == uid }) else { continue }
                
                var note = self!._notes.remove(at: index)
                note.isDeleted = true
                note.deletedDate = Date()
                
                self?._trashedNotes.append(note)
            }
            
            // Persist
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    /// Permanently delete a note.
    func deleteNote(uid: String) {
        queue.async(flags: .barrier) { [weak self] in
            self?._trashedNotes.removeAll { $0.uid == uid }
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    /// Permanently delete multiple notes.
    func deleteNotes(uids: [String]) {
        queue.async(flags: .barrier) { [weak self] in
            self?._trashedNotes.removeAll { uids.contains($0.uid) }
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    // MARK: - Trash Mutations
    
    /// Restore a trashed note back to active notes.
    func restoreNote(uid: String) {
        queue.async(flags: .barrier) { [weak self] in
            guard let index = self?._trashedNotes.firstIndex(where: { $0.uid == uid }) else { return }
            
            var note = self!._trashedNotes.remove(at: index)
            note.isDeleted = false
            note.deletedDate = nil
            
            self?._notes.append(note)
            
            // Persist
            DefaultsUtility.shared.saveNote(note)
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    /// Restore multiple trashed notes.
    func restoreNotes(uids: [String]) {
        queue.async(flags: .barrier) { [weak self] in
            for uid in uids {
                guard let index = self?._trashedNotes.firstIndex(where: { $0.uid == uid }) else { continue }
                
                var note = self!._trashedNotes.remove(at: index)
                note.isDeleted = false
                note.deletedDate = nil
                
                self?._notes.append(note)
            }
            
            // Persist
            DefaultsUtility.shared.saveTrashedNotes(self?._trashedNotes ?? [])
            
            DispatchQueue.main.async {
                NotificationCenter.default.post(name: Self.notesDidChangeNotification)
                NotificationCenter.default.post(name: Self.trashedNotesDidChangeNotification)
            }
        }
    }
    
    // MARK: - Expiry
    
    /// Enforce 15-day auto-delete for trashed notes.
    private func enforceTrashedNoteExpiry() {
        let fifteenDaysAgo = Date().addingTimeInterval(-15 * 86400)
        _trashedNotes.removeAll { note in
            guard let deletedDate = note.deletedDate else { return false }
            return deletedDate < fifteenDaysAgo
        }
    }
}
```
