```swift
// Apollo16/NETWORK/FirebaseDBManager.swift - Implement

class FirebaseDBManager {
    
    // MARK: - Note Upload
    func addANote(note: Note) {
        guard let userId = FirebaseAuth.Auth.auth().currentUser?.uid else {
            print("User not authenticated")
            return
        }
        
        let db = Firestore.firestore()
        do {
            try db.collection("users").document(userId)
                .collection("notes").document(note.uid)
                .setData(from: note)
        } catch {
            print("Error uploading note: \(error)")
        }
    }
    
    // MARK: - Note Fetch
    func getNotes(completion: @escaping ([Note]) -> Void) {
        guard let userId = FirebaseAuth.Auth.auth().currentUser?.uid else {
            completion([])
            return
        }
        
        let db = Firestore.firestore()
        db.collection("users").document(userId)
            .collection("notes")
            .addSnapshotListener { snapshot, error in
                guard let documents = snapshot?.documents else {
                    print("Error fetching notes: \(error?.localizedDescription ?? "Unknown")")
                    completion([])
                    return
                }
                
                let notes = documents.compactMap { doc -> Note? in
                    try? doc.data(as: Note.self)
                }
                
                completion(notes)
            }
    }
    
    // MARK: - Offline Queue
    private var offlineQueue: [Note] = []
    
    func queueOfflineNote(_ note: Note) {
        offlineQueue.append(note)
        saveToDisk()
    }
    
    func syncOfflineQueue() {
        for note in offlineQueue {
            addANote(note: note)
        }
        offlineQueue.removeAll()
    }
}
```
