import AVFoundation

protocol AudioRecordingManagerDelegate: AnyObject {
    func recordingDidStart()
    func recordingDidStop(duration: TimeInterval)
    func recordingDidFail(error: Error)
}

class AudioRecordingManager: NSObject, AVAudioRecorderDelegate {
    weak var delegate: AudioRecordingManagerDelegate?
    
    private var audioRecorder: AVAudioRecorder?
    private let recordingPath = NSTemporaryDirectory() + "recording.m4a"
    
    func checkPermissionAndStartRecording() {
        AVAudioSession.sharedInstance().requestRecordPermission { granted in
            if granted {
                self.startRecording()
            } else {
                self.delegate?.recordingDidFail(error: RecordingError.permissionDenied)
            }
        }
    }
    
    private func startRecording() {
        // Implementation: set up AVAudioRecorder, start recording
        delegate?.recordingDidStart()
    }
    
    func stopRecording() -> Song? {
        audioRecorder?.stop()
        let duration = audioRecorder?.currentTime ?? 0
        delegate?.recordingDidStop(duration: duration)
        
        return Song(
            title: "Recording \(Date())",
            url: recordingPath,
            duration: Int(duration),
            isDropbox: false
        )
    }
}

enum RecordingError: Error {
    case permissionDenied
    case recordingFailed(String)
}

        currentSong = song
        
        let url: URL

            // Handle Dropbox URL via DropboxManager

        } else {
            url = URL(fileURLWithPath: song.url)

            audioPlayer = try AVAudioPlayer(contentsOf: url)
            audioPlayer?.delegate = self
            audioPlayer?.play()
            setupRemoteControls()
            updateNowPlaying()
            delegate?.playbackDidStart()

            delegate?.playbackDidFail(error: error)
        }

        audioPlayer?.pause()

import MediaPlayer

protocol MusicLibraryManagerDelegate: AnyObject {
    func musicLibraryDidSelectSong(_ song: Song)
    func musicLibrarySelectionCancelled()
}

class MusicLibraryManager: NSObject, MPMediaPickerControllerDelegate {
    weak var delegate: MusicLibraryManagerDelegate?
    
    func presentMusicPicker(from viewController: UIViewController) {
        let picker = MPMediaPickerController(mediaTypes: .music)
        picker.delegate = self
        viewController.present(picker, animated: true)
    }
    
    // MARK: - MPMediaPickerControllerDelegate
    
    func mediaPicker(_ mediaPicker: MPMediaPickerController,
                    didPickMediaItems mediaItemCollection: MPMediaItemCollection) {
        if let item = mediaItemCollection.items.first {
            let song = Song(
                title: item.title ?? "Unknown",
                url: item.assetURL?.absoluteString ?? "",
                duration: Int(item.playbackDuration),
                isDropbox: false
            )
            delegate?.musicLibraryDidSelectSong(song)
        }
        mediaPicker.dismiss(animated: true)
    }
    
    func mediaPickerDidCancel(_ mediaPicker: MPMediaPickerController) {
        delegate?.musicLibrarySelectionCancelled()
        mediaPicker.dismiss(animated: true)
    }
}

### Manager 5: TextUndoManager
**New File:** `Apollo16/MANAGERS/Text/TextUndoManager.swift`

```swift
class TextUndoManager {
    private var undoStack: [String] = []
    private var redoStack: [String] = []
    private let maxStackSize = 50
    
    func recordState(_ text: String) {
        undoStack.append(text)
        if undoStack.count > maxStackSize {
            undoStack.removeFirst()
        }
        redoStack.removeAll()
    }
    
    func undo() -> String? {
        guard !undoStack.isEmpty else { return nil }
        let current = undoStack.removeLast()
        redoStack.append(current)
        return undoStack.last
    }
    
    func redo() -> String? {
        guard !redoStack.isEmpty else { return nil }
        let current = redoStack.removeLast()
        undoStack.append(current)
        return current
    }
    
    var canUndo: Bool { !undoStack.isEmpty }
    var canRedo: Bool { !redoStack.isEmpty }
}
```

class AddNote: UIViewController {
    // MARK: - Properties
    @IBOutlet weak var textView: UITextView!
    @IBOutlet weak var btn_undo: UIButton!
    @IBOutlet weak var btn_redo: UIButton!
    
    var currentNote: Note!
    
    // Managers
    private let audioRecordingManager = AudioRecordingManager()
    private let audioPlaybackManager = AudioPlaybackManager()
    private let musicLibraryManager = MusicLibraryManager()
    private let dropboxManager = DropboxManager.shared
    private let textUndoManager = TextUndoManager()
    private let rhymingWordsService = RhymingWordsService()
    
    // MARK: - Lifecycle
    
    override func viewDidLoad() {
        super.viewDidLoad()
        
        setupDelegates()
        setupObservers()
        loadNote()
    }
    
    deinit {
        NotificationCenter.default.removeObserver(self)
    }
    
    // MARK: - Setup
    
    private func setupDelegates() {
        audioRecordingManager.delegate = self
        audioPlaybackManager.delegate = self
        musicLibraryManager.delegate = self
        dropboxManager.delegate = self
        rhymingWordsService.delegate = self
    }
    
    private func setupObservers() {
        NotificationCenter.default.addObserver(
            self,
            selector: #selector(resignedActiveState),
            name: UIApplication.didResignActiveNotification,
            object: nil
        )
    }
    
    // MARK: - Note Management
    
    private func loadNote() {
        // Load currentNote
        textView.text = currentNote.content
    }
    
    @objc func resignedActiveState() {
        saveCurrentNoteIfNeeded()
    }
    
    private func saveCurrentNoteIfNeeded() {
        guard textView.text != currentNote.content else { return }
        
        currentNote.content = textView.text
        NoteRepository.shared.saveNote(currentNote)
    }
    
    @IBAction func onBackPressed(_ sender: Any) {
        saveCurrentNoteIfNeeded()
        navigationController?.popViewController(animated: true)
    }
    
    // MARK: - Audio Recording
    
    @IBAction func btn_recordTapped(_ sender: Any) {
        audioRecordingManager.checkPermissionAndStartRecording()
    }
    
    // MARK: - Audio Playback
    
    @IBAction func btn_playTapped(_ sender: Any) {
        // Get song and play

    }
    
    // MARK: - Music Library
    
    @IBAction func btn_addMusicTapped(_ sender: Any) {
        musicLibraryManager.presentMusicPicker(from: self)
    }
    
    // MARK: - Dropbox

        dropboxManager.presentDropboxAuthFlow(from: self)

            textView.text = nextState
            updateUndoRedoButtons()
        }
    }
    
    private func updateUndoRedoButtons() {
        btn_undo.isEnabled = textUndoManager.canUndo
        btn_redo.isEnabled = textUndoManager.canRedo
    }
    
    // MARK: - Rhyming Words

        let searchVC = SearchTableViewController()
        searchVC.delegate = self

    }
