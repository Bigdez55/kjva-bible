## SECTION 1: POSITION SIZING — REPLACING FIXED 2%

### Mathematical Foundation

**Full Kelly:** `f* = (W * R - L) / R`
Where:
- `W` = win rate (fraction of trades that profit)
- `L` = loss rate (1 - W)
- `R` = win/loss ratio (avg win / avg loss)

**Fractional Kelly at 0.25x** (used here to limit ruin probability):
`f_fractional = 0.25 * f*`

**Volatility scaling:** `f_vol = f_fractional * (target_vol / actual_vol)`
Where `target_vol = 0.15` (15% annualized) and `actual_vol` = EWMA vol on 20 bars.

**ATR-based sizing** (complementary):
`position_size = risk_amount / (ATR_multiplier * ATR)`
Where `risk_amount = portfolio_value * 0.01` (risk 1% per trade), `ATR_multiplier = 2.5`.

### Python Pseudocode

```python
from dataclasses import dataclass
from typing import Optional
import numpy as np

@dataclass
class SizingContext:
    portfolio_value: float       # Total portfolio NAV
    current_price: float         # Entry price
    atr: float                   # ATR(14) in dollars
    win_rate: float              # Historical win rate [0,1], default 0.5 if unknown
    win_loss_ratio: float        # avg_win / avg_loss, default 1.5 if unknown
    ai_confidence: float         # Calibrated AI confidence [0.55, 1.0]
    current_drawdown_pct: float  # Drawdown from HWM as positive fraction
    ewma_vol_annual: float       # EWMA annualized volatility
    max_position_pct: float = 0.15  # Hard cap: 15% of portfolio


def kelly_position_size(ctx: SizingContext) -> float:
    """
    Compute fractional Kelly position size with volatility normalization,
    drawdown recovery scaling, and AI conviction scaling.

    Returns number of shares to buy/sell.
    """
    # --- Step 1: Full Kelly fraction ---
    # f* = (W*R - L) / R where W=win_rate, L=1-W, R=win_loss_ratio
    W = max(0.01, min(0.99, ctx.win_rate))
    R = max(0.1, ctx.win_loss_ratio)
    L = 1.0 - W
    full_kelly = (W * R - L) / R
    full_kelly = max(0.0, full_kelly)  # Kelly must be non-negative to trade

    # --- Step 2: Fractional Kelly (0.25x safety factor) ---
    # This is the canonical "quarter Kelly" used by professional systematic funds
    # It reduces position size by 75% to account for parameter estimation error
    fractional_kelly = 0.25 * full_kelly

    # --- Step 3: Volatility normalization ---
    # Scale size down when vol is elevated to maintain constant dollar risk
    # Target 15% annualized portfolio volatility (institutional baseline)
    TARGET_VOL = 0.15
    actual_vol = max(0.05, ctx.ewma_vol_annual)  # Floor at 5% to prevent division by zero
    vol_scalar = min(TARGET_VOL / actual_vol, 1.5)  # Cap at 1.5x (never lever up excessively)

    # --- Step 4: ATR-based risk anchor ---
    # Risk exactly 1% of portfolio per trade, anchored to ATR
    # This is the "fixed fractional" method used as a floor check
    RISK_PCT_PER_TRADE = 0.01
    dollar_risk = ctx.portfolio_value * RISK_PCT_PER_TRADE
    if ctx.atr > 0:
        atr_shares = dollar_risk / (2.5 * ctx.atr)  # 2.5x ATR stop distance
    else:
        atr_shares = 0.0

    # --- Step 5: Drawdown recovery protocol ---
    # Reduce sizing 25% per 5% drawdown band from high-water mark
    # 0-5% DD: 100% size | 5-10% DD: 75% | 10-15% DD: 50% | >15% DD: 25%
    drawdown_scalar = 1.0
    if ctx.current_drawdown_pct >= 0.05:
        bands = min(int(ctx.current_drawdown_pct / 0.05), 3)
        drawdown_scalar = 1.0 - (bands * 0.25)
    drawdown_scalar = max(0.25, drawdown_scalar)

    # --- Step 6: AI conviction scaling ---
    # Scale linearly from 0.55 (minimum gate threshold) to 1.0 (max confidence)
    # At 0.55 confidence: 0.55x size multiplier
    # At 1.00 confidence: 1.00x size multiplier
    MIN_GATE_CONF = 0.55
    conviction_scalar = MIN_GATE_CONF + (1.0 - MIN_GATE_CONF) * (
        (ctx.ai_confidence - MIN_GATE_CONF) / (1.0 - MIN_GATE_CONF)
    ) if ctx.ai_confidence > MIN_GATE_CONF else MIN_GATE_CONF

    # --- Step 7: Combine all factors ---
    # Use Kelly fraction to determine base portfolio percentage
    base_pct = fractional_kelly * vol_scalar * drawdown_scalar * conviction_scalar

    # --- Step 8: Hard constraints ---
    base_pct = min(base_pct, ctx.max_position_pct)   # Never > 15% of portfolio
    base_pct = max(base_pct, 0.001)                   # Never < 0.1% (minimum meaningful)

    # --- Step 9: Translate to shares ---
    position_value = ctx.portfolio_value * base_pct
    kelly_shares = position_value / ctx.current_price

    # --- Step 10: Take the minimum of Kelly shares and ATR-anchored shares ---
    # ATR method is a hard risk anchor; Kelly method determines optimal fraction
    # Minimum of the two prevents taking more risk than either method allows
    if atr_shares > 0:
        final_shares = min(kelly_shares, atr_shares)
    else:
        final_shares = kelly_shares

    return round(final_shares, 6)


def compute_ewma_vol(returns: list[float], span: int = 20) -> float:
    """Compute EWMA annualized volatility from daily returns."""
    if len(returns) < 5:
        return 0.20  # Default 20% vol if insufficient data
    r = np.array(returns, dtype=float)
    # EWMA variance with decay factor lambda = 1 - 2/(span+1)
    lam = 1.0 - (2.0 / (span + 1))
    weights = np.array([(1 - lam) * (lam ** i) for i in range(len(r))])[::-1]
    weights /= weights.sum()
    ewma_var = np.dot(weights, (r - r.mean()) ** 2)
    return float(np.sqrt(ewma_var * 252))  # Annualize
```

### 2A: ADX Filter for Mean Reversion

```python
def compute_adx(df: pd.DataFrame, period: int = 14) -> float:
    """
    Compute ADX(14). Returns the scalar ADX value for the last bar.

    ADX < 25: market is ranging (mean reversion strategies WORK here)
    ADX > 25: market is trending (mean reversion strategies FAIL here)
    ADX > 40: strong trend (use only momentum strategies)
    """
    high, low, close = df['high'], df['low'], df['close']

    # True Range
    tr = pd.concat([
        high - low,
        (high - close.shift(1)).abs(),
        (low - close.shift(1)).abs()
    ], axis=1).max(axis=1)

    # Directional movements
    dm_plus = high.diff()
    dm_minus = -low.diff()
    dm_plus = dm_plus.where((dm_plus > dm_minus) & (dm_plus > 0), 0.0)
    dm_minus = dm_minus.where((dm_minus > dm_plus) & (dm_minus > 0), 0.0)

    # Smoothed averages
    atr = tr.ewm(span=period, adjust=False).mean()
    di_plus = 100 * dm_plus.ewm(span=period, adjust=False).mean() / atr
    di_minus = 100 * dm_minus.ewm(span=period, adjust=False).mean() / atr
    dx = 100 * (di_plus - di_minus).abs() / (di_plus + di_minus).replace(0, np.nan)
    adx = dx.ewm(span=period, adjust=False).mean()

    return float(adx.iloc[-1])


def should_trade_mean_reversion(df: pd.DataFrame, adx_threshold: float = 25.0) -> bool:
    """
    Gate: Only allow mean reversion trades when market is ranging.
    Returns True if it is safe to enter a mean reversion trade.
    """
    adx = compute_adx(df)
    return adx < adx_threshold
```

### 2B: Regime Detection

```python
from enum import Enum

class MarketRegime(str, Enum):
    BULL_LOW_VOL = "bull_low_vol"
    BULL_HIGH_VOL = "bull_high_vol"
    BEAR_LOW_VOL = "bear_low_vol"
    BEAR_HIGH_VOL = "bear_high_vol"
    SIDEWAYS = "sideways"


def detect_regime(
    spy_closes: list[float],      # 60+ days of SPY closes
    vix_level: Optional[float],   # Current VIX (or None if unavailable)
) -> MarketRegime:
    """
    4-quadrant regime detection using SPY trend + VIX level.

    Trend: SPY 50-day SMA slope (positive = bull, negative = bear)
    Volatility: VIX threshold at 20 (low/high split)
    Sideways: SPY range < 5% over 20 days AND ADX < 20

    This function runs once per loop iteration and caches result for 1 hour.
    """
    arr = np.array(spy_closes, dtype=float)
    if len(arr) < 50:
        return MarketRegime.SIDEWAYS  # Insufficient data, conservative default

    # 50-day SMA and its slope over last 10 days
    sma50 = np.mean(arr[-50:])
    sma50_prev = np.mean(arr[-60:-10]) if len(arr) >= 60 else sma50
    sma_slope = (sma50 - sma50_prev) / sma50_prev  # fractional slope

    # 20-day range as fraction of price
    range_20d = (arr[-20:].max() - arr[-20:].min()) / arr[-20:].mean()

    # Sideways detection: narrow range AND flat trend
    if range_20d < 0.05 and abs(sma_slope) < 0.005:
        return MarketRegime.SIDEWAYS

    # VIX threshold
    HIGH_VOL_VIX = 20.0
    is_high_vol = (vix_level is not None and vix_level >= HIGH_VOL_VIX) or range_20d > 0.08

    is_bull = sma_slope > 0

    if is_bull and not is_high_vol:
        return MarketRegime.BULL_LOW_VOL
    elif is_bull and is_high_vol:
        return MarketRegime.BULL_HIGH_VOL
    elif not is_bull and not is_high_vol:
        return MarketRegime.BEAR_LOW_VOL
    else:
        return MarketRegime.BEAR_HIGH_VOL
```

### 2D: GARCH/EWMA Volatility Forecast

The existing `FeatureEngineering` already computes ATR. The missing piece is a forward-looking vol estimate for position sizing.

```python
def forecast_vol_ewma(returns: np.ndarray, horizon: int = 1) -> float:
    """
    EWMA volatility forecast for N days ahead.

    Uses RiskMetrics decay factor lambda=0.94 (daily, industry standard).
    Returns annualized vol forecast.

    This replaces the realized ATR-based vol with a forward-looking estimate
    that adapts faster to regime changes (EWMA responds in ~8 bars vs ATR 14 bars).
    """
    lam = 0.94
    if len(returns) < 5:
        return 0.20

    # EWMA variance update: sigma^2_t = lambda * sigma^2_{t-1} + (1-lambda) * r^2_{t-1}
    var = float(np.var(returns[:5]))  # Bootstrap from first 5 observations
    for r in returns[5:]:
        var = lam * var + (1 - lam) * (r ** 2)

    # Scale forward: for horizon > 1 day, multiply variance by horizon (square root of time rule)
    # This is only valid under IID returns -- for intraday, use realized variance instead
    forecasted_var = var * horizon
    return float(np.sqrt(forecasted_var * 252))  # Annualize
```

### 3A: ATR-Based Dynamic Stops

```python
def compute_atr_stops(
    entry_price: float,
    atr: float,
    action: str,  # "buy" or "sell"
    atr_multiplier: float = 2.5,
    max_loss_pct: float = 0.08,  # Hard stop at -8% from entry
) -> dict:
    """
    Compute ATR-based stop loss and take profit levels.

    ATR multiplier of 2.5 is the institutional baseline:
    - Below 2.0: too tight, gets stopped out by noise
    - Above 3.0: too wide, damages risk/reward ratio
    - 2.5: optimal balance per Van Tharp / Kaufman research

    Hard stop at -8% regardless of ATR width (prevents gap risk from
    wiping > 8% on a single position even in extreme conditions).
    """
    atr_stop_distance = atr_multiplier * atr

    if action == "buy":
        atr_stop = entry_price - atr_stop_distance
        hard_stop = entry_price * (1 - max_loss_pct)
        stop_loss = max(atr_stop, hard_stop)  # Use tighter (higher) stop for longs
        take_profit = entry_price + (atr_stop_distance * 2.0)  # 2:1 reward/risk
    else:  # sell/short
        atr_stop = entry_price + atr_stop_distance
        hard_stop = entry_price * (1 + max_loss_pct)
        stop_loss = min(atr_stop, hard_stop)  # Use tighter (lower) stop for shorts
        take_profit = entry_price - (atr_stop_distance * 2.0)

    risk_pct = abs(entry_price - stop_loss) / entry_price
    reward_pct = abs(take_profit - entry_price) / entry_price

    return {
        "stop_loss": round(stop_loss, 4),
        "take_profit": round(take_profit, 4),
        "atr_stop": round(atr_stop, 4),
        "hard_stop": round(hard_stop, 4),
        "risk_pct": round(risk_pct, 4),
        "reward_risk_ratio": round(reward_pct / risk_pct, 2) if risk_pct > 0 else 0.0,
    }
```

TIME_BASED_EXITS = {

def check_time_stop(
    entry_date: date,
    current_pnl_pct: float,
    strategy_name: str,
    max_hold_days: Optional[int] = None,
) -> bool:
    """
    Returns True if the position should be exited due to time-based stop.

    Logic: If a mean reversion trade hasn't worked in 5 days, the thesis
    is wrong — the price is NOT reverting, it's trending. Exit and redeploy capital.
    """
    hold_days = (date.today() - entry_date).days
    allowed_days = max_hold_days or TIME_BASED_EXITS.get(strategy_name, 10)

    # Only exit on time stop if position is not profitable
    # If profitable: let it run (winners take care of themselves)
    if hold_days >= allowed_days and current_pnl_pct <= 0.0:
        return True
    return False

### 3C: Trailing Stops for Momentum Strategies

```python
class TrailingStop:
    """
    ATR-based trailing stop that ratchets up behind a winning position.

    Activate only for momentum strategies where the thesis is:
    "the trend continues until it doesn't"

    Usage:
        ts = TrailingStop(entry_price=100.0, atr=2.5, multiplier=3.0)
        ts.update(current_price=108.0)  # Ratchet stop up
        if ts.is_triggered(current_price=104.0):
            execute_exit()
    """

    def __init__(self, entry_price: float, atr: float, multiplier: float = 3.0):
        self.entry_price = entry_price
        self.atr = atr
        self.multiplier = multiplier
        self.high_water_mark = entry_price
        self.stop_price = entry_price - (multiplier * atr)

    def update(self, current_price: float) -> None:
        """Move the stop up if price makes new high (for long positions)."""
        if current_price > self.high_water_mark:
            self.high_water_mark = current_price
            self.stop_price = current_price - (self.multiplier * self.atr)

    def is_triggered(self, current_price: float) -> bool:
        """Returns True if trailing stop has been hit."""
        return current_price <= self.stop_price

    @property
    def current_profit_pct(self) -> float:
        return (self.high_water_mark - self.entry_price) / self.entry_price
```

### Strategy 13: Gap Fill (Fade the Gap)

### Proposed Ensemble Architecture

```python
from dataclasses import dataclass
from typing import Optional
from enum import Enum

class DisagreementResolution(str, Enum):
    VETO_ON_STRONG_DISAGREEMENT = "veto"  # All must agree directionally
    WEIGHTED_AVERAGE = "weighted_avg"      # Current approach
    CONFIDENCE_WEIGHTED = "conf_weighted"  # Higher confidence signal wins more
    REGIME_ADAPTIVE = "regime_adaptive"   # Regime determines weight allocation


@dataclass
class SignalInput:
    action: str  # "buy" | "sell" | "hold"
    confidence: float
    source: str


def ensemble_signal(
    rule_signal: Optional[SignalInput],
    llm_signal: Optional[SignalInput],
    drl_signal: Optional[SignalInput],
    regime: MarketRegime,
    resolution: DisagreementResolution = DisagreementResolution.REGIME_ADAPTIVE,
) -> dict:
    """
    Combine three independent signal sources into a single trading decision.

    REGIME-ADAPTIVE WEIGHT TABLE:

    Regime          | Rule  | LLM  | DRL  | Rationale
    ----------------|-------|------|------|------------------------------------------
    BULL_LOW_VOL    | 0.30  | 0.40 | 0.30 | All signals reliable in calm bull market
    BULL_HIGH_VOL   | 0.50  | 0.30 | 0.20 | Vol spike: rule-based more stable
    BEAR_LOW_VOL    | 0.40  | 0.35 | 0.25 | Bear: rules and DRL better than LLM
    BEAR_HIGH_VOL   | 0.60  | 0.20 | 0.20 | Crisis: rules dominate, LLM hallucinate
    SIDEWAYS        | 0.35  | 0.40 | 0.25 | Sideways: LLM context helps most

    DISAGREEMENT HANDLING:
    Strong disagreement = any two sources are directionally opposite with > 0.7 confidence.
    Under strong disagreement: HOLD (do not trade). The signal is ambiguous.

    Why veto on strong disagreement?
    Expected value analysis:
      - If two models disagree at 70%+ confidence, the true expected direction is near 50/50
      - A 50/50 trade after commission = negative expected value
      - Better to wait for convergence
    """
    REGIME_WEIGHTS = {
        MarketRegime.BULL_LOW_VOL:  {"rule": 0.30, "llm": 0.40, "drl": 0.30},
        MarketRegime.BULL_HIGH_VOL: {"rule": 0.50, "llm": 0.30, "drl": 0.20},
        MarketRegime.BEAR_LOW_VOL:  {"rule": 0.40, "llm": 0.35, "drl": 0.25},
        MarketRegime.BEAR_HIGH_VOL: {"rule": 0.60, "llm": 0.20, "drl": 0.20},
        MarketRegime.SIDEWAYS:      {"rule": 0.35, "llm": 0.40, "drl": 0.25},
    }

    weights = REGIME_WEIGHTS.get(regime, {"rule": 0.40, "llm": 0.40, "drl": 0.20})

    signals = {
        "rule": rule_signal,
        "llm": llm_signal,
        "drl": drl_signal,
    }

    # Redistribute weight from missing signals proportionally
    active = {k: v for k, v in signals.items() if v is not None}
    if not active:
        return {"action": "hold", "confidence": 0.0, "reasoning": "no_signals", "sources": []}

    missing_weight = sum(weights[k] for k in signals if signals[k] is None)
    if missing_weight > 0:
        active_total = sum(weights[k] for k in active)
        weights = {
            k: weights[k] + (missing_weight * weights[k] / active_total)
            if k in active else 0.0
            for k in weights
        }

    # Normalize
    total_weight = sum(weights.values())
    weights = {k: v / total_weight for k, v in weights.items()}

    # --- Strong disagreement veto ---
    actions = {k: v.action for k, v in active.items()}
    confs = {k: v.confidence for k, v in active.items()}

    directional_active = {k: a for k, a in actions.items() if a != "hold"}
    if len(directional_active) >= 2:
        unique_directions = set(directional_active.values())
        if len(unique_directions) > 1:
            # Check if any two are strongly opposing
            strong_disagree_pairs = [
                (k1, k2)
                for k1, k2 in [
                    (a, b) for a in directional_active for b in directional_active if a != b
                ]
                if directional_active[k1] != directional_active[k2]
                   and confs[k1] > 0.70
                   and confs[k2] > 0.70
            ]
            if strong_disagree_pairs:
                return {
                    "action": "hold",
                    "confidence": 0.0,
                    "reasoning": f"strong_disagreement_veto: {strong_disagree_pairs}",
                    "sources": list(active.keys()),
                    "weights_used": weights,
                }

    # --- Weighted vote ---
    ACTION_SCORE = {"buy": 1.0, "sell": -1.0, "hold": 0.0}

    weighted_vote = sum(
        weights.get(k, 0) * ACTION_SCORE.get(v.action, 0) * v.confidence
        for k, v in active.items()
    )

    weighted_conf = sum(
        weights.get(k, 0) * v.confidence
        for k, v in active.items()
    )

    BUY_THRESHOLD = 0.15
    SELL_THRESHOLD = -0.15

    if weighted_vote >= BUY_THRESHOLD:
        action = "buy"
    elif weighted_vote <= SELL_THRESHOLD:
        action = "sell"
    else:
        action = "hold"

    return {
        "action": action,
        "confidence": round(abs(weighted_vote), 4),
        "raw_vote": round(weighted_vote, 4),
        "weighted_confidence": round(weighted_conf, 4),
        "weights_used": weights,
        "sources": list(active.keys()),
        "reasoning": f"regime={regime.value} vote={weighted_vote:.3f}",
    }
```

### Walk-Forward Validation Framework

```python
from datetime import date, timedelta
from typing import List, Tuple

def walk_forward_periods(
    start: date,
    end: date,
    train_months: int = 6,
    test_months: int = 1,
) -> List[Tuple[Tuple[date, date], Tuple[date, date]]]:
    """
    Generate (train_period, test_period) tuples for walk-forward validation.

    Example: 2022-01-01 to 2024-12-31, 6-month train, 1-month test:
      Train: Jan-Jun 2022 → Test: Jul 2022
      Train: Feb-Jul 2022 → Test: Aug 2022
      ... (rolling)

    This is the ONLY valid method for strategy validation.
    In-sample optimization followed by testing on the SAME data = look-ahead bias.
    """
    periods = []
    cursor = start
    while True:
        train_end = cursor + timedelta(days=30 * train_months)
        test_start = train_end
        test_end = test_start + timedelta(days=30 * test_months)
        if test_end > end:
            break
        periods.append(((cursor, train_end), (test_start, test_end)))
        cursor += timedelta(days=30)  # Advance by 1 month
    return periods

        commission_rate=0.001,  # 0.1% Alpaca
        slippage_rate=0.0005,   # 5bps slippage
        position_sizing="kelly",
        position_size=0.02,
    ))

    old_results = {}
    new_results = {}

    for symbol in symbols:
        old_result = await engine.run(strategy_old, symbol, test_period)
        new_result = await engine.run(strategy_new, symbol, test_period)
        old_results[symbol] = old_result
        new_results[symbol] = new_result

    # Aggregate metrics
    return {
        "old": aggregate_backtest_results(old_results),
        "new": aggregate_backtest_results(new_results),
        "deploy_recommended": (
            new_results["sharpe"] > old_results["sharpe"] + 0.2
            and new_results["max_drawdown"] < 0.15
            and new_results["total_trades"] >= 100
        )
    }
```
